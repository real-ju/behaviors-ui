Textarea 文本域 ​

基本使用 ​

html
<be-textarea
  class="textarea1"
  v-model="content"
  placeholder="请输入"
  placeholder-class="textarea1-placeholder"
>
</be-textarea>
ts
import { ref } from 'vue';

const content = ref('');
scss
.textarea1 {
  background-color: #f2f2f2;
  font-size: 14px;

  .textarea1-placeholder {
    color: #bbbbbb;
  }
}
